    "${CMAKE_CXX_COMPILE_FEATURES}"
    CACHE STRING "C++ compile features" FORCE)

    "${CMAKE_CXX_COMPILER_ID}"
    CACHE STRING "C++ compiler ID" FORCE)

    "${CMAKE_CXX_COMPILER_VERSION}"
    CACHE STRING "C++ compiler version" FORCE)
